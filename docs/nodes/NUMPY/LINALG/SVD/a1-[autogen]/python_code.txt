from flojoy import OrderedPair, flojoy
import numpy.linalg


@flojoy(node_type='default')
def SVD(
	default: OrderedPair,
	full_matrices: bool = True,
	compute_uv: bool = True,
	hermitian: bool = False,
	) -> OrderedPair:
	''''''
	result = OrderedPair(
		x=default.x,
		y=numpy.linalg.svd(
			a=default.y,
			full_matrices=full_matrices,
			compute_uv=compute_uv,
			hermitian=hermitian,
		)
	)
	return result
