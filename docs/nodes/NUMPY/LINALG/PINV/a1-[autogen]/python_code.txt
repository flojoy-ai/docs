from flojoy import OrderedPair, flojoy
import numpy.linalg


@flojoy(node_type='default')
def PINV(
	default: OrderedPair,
	rcond: float = 1e-15,
	hermitian: bool = False,
	) -> OrderedPair:
	''''''
	result = OrderedPair(
		x=default.x,
		y=numpy.linalg.pinv(
			a=default.y,
			rcond=rcond,
			hermitian=hermitian,
		)
	)
	return result
