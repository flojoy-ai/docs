from flojoy import DataContainer, flojoy
import numpy.linalg


@flojoy
def PINV(dc, params):
    
    return DataContainer(
        x=dc[0].y,
        y=numpy.linalg.pinv(
            a=dc[0].y,
            rcond=(float(params["rcond"]) if params["rcond"] != "" else None),
            hermitian=(
                bool(params["hermitian"]) if params["hermitian"] != "" else None
            ),
        ),
    )
