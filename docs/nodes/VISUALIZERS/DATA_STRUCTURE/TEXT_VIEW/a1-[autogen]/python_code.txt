import pprint
from flojoy import flojoy, TextBlob, Boolean


@flojoy
def TEXT_VIEW(default: TextBlob | Boolean, prettify: bool = False) -> TextBlob:
    

    match default:
        case TextBlob():
            s = default.text_blob

            if prettify:
                s = pprint.pformat(default.text_blob)

            return TextBlob(s)
        case Boolean():
            if default.b:
                return TextBlob("True")
            
            return TextBlob("False")
