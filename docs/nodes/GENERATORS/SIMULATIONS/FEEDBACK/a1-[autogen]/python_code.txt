from typing import Any, Optional

from flojoy import (
    DataContainer,
    JobResultBuilder,
    NodeReference,
    flojoy,
    get_job_result,
)


@flojoy
def FEEDBACK(
    referred_node: NodeReference,
    default: Optional[DataContainer] = None,
) -> Any:
    

    result = get_job_result(referred_node.ref)
    if result:
        return result
    else:
        return (
            JobResultBuilder()
            .from_inputs([default] if default else [])
            .flow_to_directions(["default"])
            .build()
        )
