from flojoy import OrderedPair, flojoy, Matrix, Scalar
import numpy as np


import scipy.signal


@flojoy(node_type="default")
def DECIMATE(
    default: OrderedPair | Matrix,
    q: int,
    n: int,
    ftype: str = "iir",
    axis: int = -1,
    zero_phase: bool = True,
) -> OrderedPair | Matrix | Scalar:
    

    result = OrderedPair(
        m=scipy.signal.decimate(
            x=default.y,
            q=q,
            n=n,
            ftype=ftype,
            axis=axis,
            zero_phase=zero_phase,
        )
    )

    return result
