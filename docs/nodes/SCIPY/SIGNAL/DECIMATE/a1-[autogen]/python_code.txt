from flojoy import OrderedPair, flojoy
import scipy.signal


@flojoy(node_type='default')
def DECIMATE(
	default: OrderedPair,
	q: int,
	n: int,
	ftype: str = 'iir',
	axis: int = -1,
	zero_phase: bool = True,
	) -> OrderedPair:
	''''''
	result = OrderedPair(
		x=default.x,
		y=scipy.signal.decimate(
			x=default.y,
			q=q,
			n=n,
			ftype=ftype,
			axis=axis,
			zero_phase=zero_phase,
		)
	)
	return result
