from flojoy import OrderedPair, flojoy
import scipy.signal


@flojoy(node_type='default')
def ARGRELMIN(
	default: OrderedPair,
	axis: int = 0,
	order: int = 1,
	mode: str = 'clip',
	) -> OrderedPair:
	''''''
	result = OrderedPair(
		x=default.x,
		y=scipy.signal.argrelmin(
			data=default.y,
			axis=axis,
			order=order,
			mode=mode,
		)
	)
	return result
