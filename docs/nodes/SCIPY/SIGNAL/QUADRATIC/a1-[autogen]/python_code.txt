from flojoy import OrderedPair, flojoy, Matrix, Scalar
import numpy as np
from collections import namedtuple
from typing import Literal

import scipy.signal


@flojoy
def QUADRATIC(
    default: OrderedPair | Matrix,
) -> OrderedPair | Matrix | Scalar:
    

    result = OrderedPair(
        x=default.x,
        y=scipy.signal.quadratic(
            x=default.y,
        ),
    )

    return result
