from flojoy import OrderedPair, flojoy
import scipy.signal


@flojoy(node_type='default')
def HILBERT(
	default: OrderedPair,
	N: int,
	axis: int = -1,
	) -> OrderedPair:
	''''''
	result = OrderedPair(
		x=default.x,
		y=scipy.signal.hilbert(
			x=default.y,
			N=N,
			axis=axis,
		)
	)
	return result
