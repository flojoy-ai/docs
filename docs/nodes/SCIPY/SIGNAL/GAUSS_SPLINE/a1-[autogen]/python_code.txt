from flojoy import OrderedPair, flojoy, Matrix, Scalar
import numpy as np
from collections import namedtuple
from typing import Literal

import scipy.signal


@flojoy
def GAUSS_SPLINE(
    default: OrderedPair | Matrix,
    n: int,
) -> OrderedPair | Matrix | Scalar:
    

    result = OrderedPair(
        x=default.x,
        y=scipy.signal.gauss_spline(
            x=default.y,
            n=n,
        ),
    )

    return result
