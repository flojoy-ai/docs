from flojoy import OrderedPair, flojoy
import scipy.signal


@flojoy(node_type='default')
def DETREND(
	default: OrderedPair,
	axis: int = -1,
	type: str = 'linear',
	bp: int = 0,
	overwrite_data: bool = False,
	) -> OrderedPair:
	''''''
	result = OrderedPair(
		x=default.x,
		y=scipy.signal.detrend(
			data=default.y,
			axis=axis,
			type=type,
			bp=bp,
			overwrite_data=overwrite_data,
		)
	)
	return result
