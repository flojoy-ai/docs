from flojoy import OrderedPair, flojoy, Matrix, Scalar
import numpy as np


import scipy.signal


@flojoy(node_type="default")
def ARGRELMAX(
    default: OrderedPair | Matrix,
    axis: int = 0,
    order: int = 1,
    mode: str = "clip",
) -> OrderedPair | Matrix | Scalar:
    

    result = OrderedPair(
        m=scipy.signal.argrelmax(
            data=default.y,
            axis=axis,
            order=order,
            mode=mode,
        )
    )

    return result
