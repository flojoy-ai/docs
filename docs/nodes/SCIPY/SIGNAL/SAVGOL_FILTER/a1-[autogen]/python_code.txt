from flojoy import OrderedPair, flojoy, Matrix, Scalar
import numpy as np
from collections import namedtuple
from typing import Literal

import scipy.signal


@flojoy
def SAVGOL_FILTER(
    default: OrderedPair | Matrix,
    window_length: int,
    polyorder: int,
    deriv: int = 0,
    delta: float = 1.0,
    axis: int = -1,
    mode: str = "interp",
    cval: float = 0.0,
) -> OrderedPair | Matrix | Scalar:
    

    result = OrderedPair(
        x=default.x,
        y=scipy.signal.savgol_filter(
            x=default.y,
            window_length=window_length,
            polyorder=polyorder,
            deriv=deriv,
            delta=delta,
            axis=axis,
            mode=mode,
            cval=cval,
        ),
    )

    return result
