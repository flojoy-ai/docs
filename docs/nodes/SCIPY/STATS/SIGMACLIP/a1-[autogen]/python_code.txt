from flojoy import DataContainer, flojoy
import scipy.stats


@flojoy
def SIGMACLIP(dc, params):
    
    return DataContainer(
        x=dc[0].y,
        y=scipy.stats.sigmaclip(
            a=dc[0].y,
            low=(float(params["low"]) if params["low"] != "" else None),
            high=(float(params["high"]) if params["high"] != "" else None),
        ),
    )
