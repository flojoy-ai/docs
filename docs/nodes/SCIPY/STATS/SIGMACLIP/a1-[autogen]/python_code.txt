from flojoy import OrderedPair, flojoy
import scipy.stats


@flojoy(node_type='default')
def SIGMACLIP(
	default: OrderedPair,
	low: float = 4.0,
	high: float = 4.0,
	) -> OrderedPair:
	''''''
	result = OrderedPair(
		x=default.x,
		y=scipy.stats.sigmaclip(
			a=default.y,
			low=low,
			high=high,
		)
	)
	return result
