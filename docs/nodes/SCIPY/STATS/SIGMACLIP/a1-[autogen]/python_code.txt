from flojoy import OrderedPair, flojoy, Matrix, Scalar
import numpy as np


import scipy.stats


@flojoy(node_type="default")
def SIGMACLIP(
    default: OrderedPair | Matrix,
    low: float = 4.0,
    high: float = 4.0,
) -> OrderedPair | Matrix | Scalar:
    

    result = OrderedPair(
        m=scipy.stats.sigmaclip(
            a=default.y,
            low=low,
            high=high,
        )
    )

    return result
