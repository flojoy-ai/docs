from flojoy import OrderedPair, flojoy, Matrix, Scalar
import numpy as np


import scipy.stats


@flojoy(node_type="default")
def TRIM1(
    default: OrderedPair | Matrix,
    proportiontocut: float,
    tail: str = "right",
    axis: int = 0,
) -> OrderedPair | Matrix | Scalar:
    

    result = OrderedPair(
        m=scipy.stats.trim1(
            a=default.y,
            proportiontocut=proportiontocut,
            tail=tail,
            axis=axis,
        )
    )

    return result
