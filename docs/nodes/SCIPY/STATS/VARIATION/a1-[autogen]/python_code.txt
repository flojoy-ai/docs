from flojoy import OrderedPair, flojoy
import scipy.stats


@flojoy(node_type='default')
def VARIATION(
	default: OrderedPair,
	axis: int = 0,
	nan_policy: str = 'propagate',
	ddof: int = 0,
	keepdims: bool = False,
	) -> OrderedPair:
	''''''
	result = OrderedPair(
		x=default.x,
		y=scipy.stats.variation(
			a=default.y,
			axis=axis,
			nan_policy=nan_policy,
			ddof=ddof,
			keepdims=keepdims,
		)
	)
	return result
