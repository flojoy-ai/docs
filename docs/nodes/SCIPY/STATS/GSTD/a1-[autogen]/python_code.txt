from flojoy import OrderedPair, flojoy, Matrix, Scalar
import numpy as np
from collections import namedtuple
from typing import Literal

import scipy.stats


@flojoy
def GSTD(
    default: OrderedPair | Matrix,
    axis: int = 0,
    ddof: int = 1,
) -> OrderedPair | Matrix | Scalar:
    

    result = OrderedPair(
        x=default.x,
        y=scipy.stats.gstd(
            a=default.y,
            axis=axis,
            ddof=ddof,
        ),
    )

    return result
