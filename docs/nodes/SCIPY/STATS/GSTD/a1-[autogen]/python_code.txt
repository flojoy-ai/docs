from flojoy import DataContainer, flojoy
import scipy.stats


@flojoy
def GSTD(dc, params):
    
    return DataContainer(
        x=dc[0].y,
        y=scipy.stats.gstd(
            a=dc[0].y,
            axis=(int(params["axis"]) if params["axis"] != "" else None),
            ddof=(int(params["ddof"]) if params["ddof"] != "" else None),
        ),
    )
