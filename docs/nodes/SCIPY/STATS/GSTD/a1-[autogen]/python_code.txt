from flojoy import OrderedPair, flojoy
import scipy.stats


@flojoy(node_type='default')
def GSTD(
	default: OrderedPair,
	axis: int = 0,
	ddof: int = 1,
	) -> OrderedPair:
	''''''
	result = OrderedPair(
		x=default.x,
		y=scipy.stats.gstd(
			a=default.y,
			axis=axis,
			ddof=ddof,
		)
	)
	return result
