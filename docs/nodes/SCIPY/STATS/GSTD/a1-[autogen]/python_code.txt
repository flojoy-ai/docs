from flojoy import OrderedPair, flojoy, Matrix, Scalar
import numpy as np


import scipy.stats


@flojoy(node_type="default")
def GSTD(
    default: OrderedPair | Matrix,
    axis: int = 0,
    ddof: int = 1,
) -> OrderedPair | Matrix | Scalar:
    

    result = OrderedPair(
        m=scipy.stats.gstd(
            a=default.y,
            axis=axis,
            ddof=ddof,
        )
    )

    return result
