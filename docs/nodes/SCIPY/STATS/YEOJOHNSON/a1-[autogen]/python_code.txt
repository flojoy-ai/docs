from flojoy import OrderedPair, flojoy, Matrix, Scalar
import numpy as np


import scipy.stats


@flojoy(node_type="default")
def YEOJOHNSON(
    default: OrderedPair | Matrix,
    lmbda: float,
) -> OrderedPair | Matrix | Scalar:
    

    result = OrderedPair(
        m=scipy.stats.yeojohnson(
            x=default.y,
            lmbda=lmbda,
        )
    )

    return result
