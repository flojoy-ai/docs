from flojoy import OrderedPair, flojoy
import scipy.stats


@flojoy(node_type='default')
def TMAX(
	default: OrderedPair,
	upperlimit: None or float,
	axis: int = 0,
	inclusive: bool = True,
	nan_policy: str = 'propagate',
	) -> OrderedPair:
	''''''
	result = OrderedPair(
		x=default.x,
		y=scipy.stats.tmax(
			a=default.y,
			upperlimit=upperlimit,
			axis=axis,
			inclusive=inclusive,
			nan_policy=nan_policy,
		)
	)
	return result
