from flojoy import OrderedPair, flojoy
import scipy.stats


@flojoy(node_type='default')
def TRIMBOTH(
	default: OrderedPair,
	proportiontocut: float,
	axis: int = 0,
	) -> OrderedPair:
	''''''
	result = OrderedPair(
		x=default.x,
		y=scipy.stats.trimboth(
			a=default.y,
			proportiontocut=proportiontocut,
			axis=axis,
		)
	)
	return result
