from flojoy import OrderedPair, flojoy
import scipy.stats


@flojoy(node_type='default')
def NORMALTEST(
	default: OrderedPair,
	axis: int = 0,
	nan_policy: str = 'propagate',
	) -> OrderedPair:
	''''''
	result = OrderedPair(
		x=default.x,
		y=scipy.stats.normaltest(
			a=default.y,
			axis=axis,
			nan_policy=nan_policy,
		)
	)
	return result
