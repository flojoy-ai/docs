from flojoy import OrderedPair, flojoy
import scipy.stats


@flojoy(node_type='default')
def KURTOSISTEST(
	default: OrderedPair,
	axis: int = 0,
	nan_policy: str = 'propagate',
	alternative: str = 'two-sided',
	) -> OrderedPair:
	''''''
	result = OrderedPair(
		x=default.x,
		y=scipy.stats.kurtosistest(
			a=default.y,
			axis=axis,
			nan_policy=nan_policy,
			alternative=alternative,
		)
	)
	return result
