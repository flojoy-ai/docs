from flojoy import OrderedPair, flojoy, Matrix, Scalar
import numpy as np


import scipy.stats


@flojoy(node_type="default")
def ANDERSON(
    default: OrderedPair | Matrix,
    dist: str = "norm",
) -> OrderedPair | Matrix | Scalar:
    

    result = OrderedPair(
        m=scipy.stats.anderson(
            x=default.y,
            dist=dist,
        )
    )

    return result
