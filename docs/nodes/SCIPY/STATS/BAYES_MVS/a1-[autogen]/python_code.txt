from flojoy import OrderedPair, flojoy, Matrix, Scalar
import numpy as np


import scipy.stats


@flojoy(node_type="default")
def BAYES_MVS(
    default: OrderedPair | Matrix,
    alpha: float = 0.9,
) -> OrderedPair | Matrix | Scalar:
    

    result = OrderedPair(
        m=scipy.stats.bayes_mvs(
            data=default.y,
            alpha=alpha,
        )
    )

    return result
