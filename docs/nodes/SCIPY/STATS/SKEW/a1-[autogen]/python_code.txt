from flojoy import OrderedPair, flojoy
import scipy.stats


@flojoy(node_type='default')
def SKEW(
	default: OrderedPair,
	axis: int = 0,
	bias: bool = True,
	nan_policy: str = 'propagate',
	keepdims: bool = False,
	) -> OrderedPair:
	''''''
	result = OrderedPair(
		x=default.x,
		y=scipy.stats.skew(
			a=default.y,
			axis=axis,
			bias=bias,
			nan_policy=nan_policy,
			keepdims=keepdims,
		)
	)
	return result
