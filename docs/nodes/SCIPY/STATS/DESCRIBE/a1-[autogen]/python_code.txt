from flojoy import OrderedPair, flojoy, Matrix, Scalar
import numpy as np


import scipy.stats


@flojoy(node_type="default")
def DESCRIBE(
    default: OrderedPair | Matrix,
    axis: int = 0,
    ddof: int = 1,
    bias: bool = True,
    nan_policy: str = "propagate",
) -> OrderedPair | Matrix | Scalar:
    

    result = OrderedPair(
        m=scipy.stats.describe(
            a=default.y,
            axis=axis,
            ddof=ddof,
            bias=bias,
            nan_policy=nan_policy,
        )
    )

    return result
