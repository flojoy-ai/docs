from flojoy import OrderedPair, flojoy, Matrix, Scalar
import numpy as np


import scipy.stats


@flojoy(node_type="default")
def KURTOSIS(
    default: OrderedPair | Matrix,
    axis: int = 0,
    fisher: bool = True,
    bias: bool = True,
    nan_policy: str = "propagate",
    keepdims: bool = False,
) -> OrderedPair | Matrix | Scalar:
    

    result = OrderedPair(
        m=scipy.stats.kurtosis(
            a=default.y,
            axis=axis,
            fisher=fisher,
            bias=bias,
            nan_policy=nan_policy,
            keepdims=keepdims,
        )
    )

    return result
