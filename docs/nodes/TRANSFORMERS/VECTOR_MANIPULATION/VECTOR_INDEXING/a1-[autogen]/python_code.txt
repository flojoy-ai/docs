import numpy as np
from flojoy import flojoy, Vector, Scalar


@flojoy
def VECTOR_INDEXING(
    default: Vector,
    index: int = 0,
) -> Scalar:
    

    assert (
        len(default.v) > index
    ), "The index parameter must be less than the length of the Vector."
    assert index >= 0, "The index parameter greater than zero."
    c = default.v[index]

    return Scalar(c=c)
