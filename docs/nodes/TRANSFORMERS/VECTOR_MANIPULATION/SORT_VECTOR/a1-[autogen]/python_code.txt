from flojoy import flojoy, Vector


@flojoy
def SORT_VECTOR(
    default: Vector,
    reverse: int = 0,
) -> Vector:
    
    if reverse == 0:
        return Vector(v=sorted(default.v))
    elif reverse == 1:
        return Vector(v=sorted(default.v, reverse=True))
    else:
        raise AssertionError(f"Invalid reverse value! It should be either 0 or 1 but given {reverse}")
