import numpy as np
from flojoy import flojoy, OrderedPair


@flojoy
def ADD(a: OrderedPair, b: OrderedPair) -> OrderedPair:
    
    # TODO: Add support for adding more than 2 elements
    # TODO: Support other data types
    x = a.x

    u = a.y
    v = b.y

    y = np.add(u, v)

    return OrderedPair(x=x, y=y)
